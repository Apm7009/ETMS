SELECT:-SELECT * FROM DBO.TABLENAME
DROP:-DROP TABLE TABLENAME;  

USER:-(SELECT * FROM DBO.USERDATA)
CREATE TABLE USERDATA(   
FIRSTNAME VARCHAR(50),   
LASTNAME VARCHAR(50), 
EMPLOYEEID INT,
EMAILID VARCHAR(50),   
PHONENO VARCHAR(50), 
ADDRESS VARCHAR(50),
USERNAME VARCHAR(50), 
PASSWORD VARCHAR(50),         
PRIMARY KEY (USERNAME));

FEEDBACK:-(SELECT * FROM DBO.FEEDBACK)
CREATE TABLE FEEDBACK(   
NAME VARCHAR(50),    
EMAILID VARCHAR(50),   
RATINGS INT, 
FEEDBACK VARCHAR(50),
MESSAGEDATE VARCHAR(50)); 

CONTACTUS:-(SELECT * FROM DBO.CONTACTUS) 
CREATE TABLE CONTACTUS(   
NAME VARCHAR(50), 
EMAILID VARCHAR(50),   
PHONENO VARCHAR(50), 
MESSAGE VARCHAR(50),
MESSAGEDATE VARCHAR(50)); 

CABBOOKING:-(SELECT * FROM DBO.CABBOOKING)
CREATE TABLE CABBOOKING( 
BOOKINGID INT IDENTITY(1,1),
[USER] VARCHAR(50),
PICKUPLOCATION VARCHAR(50), 
DROPOFFLOCATION VARCHAR(50),   
TIMING VARCHAR(50), 
PRIMARY KEY (BOOKINGID)); 

ADMIN:-(SELECT * FROM DBO.ADMIN)
CREATE TABLE ADMIN(   
ADMINID INT,

USERNAME VARCHAR(50), 
PASSWORD VARCHAR(50),
PRIMARY KEY (ADMINID));
